t6 = threading.Thread(target=bgp.bgpTable)

t1.start()

t1.join()
t2.join()
t3.join()
t4.join()

t5.start()
t6.start()
t7.start()
t8.start()

t5.join()
t6.join()
t7.join()
t8.join()
